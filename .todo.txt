🟢 الحالات الممكنة بلا basic trivial (يعني بلا pwd, env normal):
1. echo

echo -n hello

echo -nnnn hi

echo -n -n test

echo (no args → newline only)

echo " " (spaces only)

echo "" (empty string)

echo - (not option, just a dash)

2. cd

cd (→ HOME)

cd ~ (expansion if implemented)

cd - (back to OLDPWD)

cd path/not/exist (error case)

cd /tmp/.. (relative parent test)

3. export

export VAR=value

export VAR="with space"

export VAR1= VAR2=value2 (VAR1 invalid → test error)

export VAR+=something (not mandatory but 42 test it)

export (no args → print env sorted)

4. unset

unset VAR

unset VAR1 VAR2 (multiple)

unset (no args → does nothing)

unset PATH (edge case, ydir break f commands li b3d)

5. exit

exit (no args → exit 0)

exit 42 (numeric arg → exit 42)

exit 256 (normalize → exit 0)

exit notnumber (error → print error, not exit)

exit 1 2 (too many args → error, shell doesn’t exit)
